25 int RFM69::inviaMessaggio(const uint8_t messaggio[], uint8_t lunghezza, uint8_t intestazione, bool insisti) {
26 
27  // la radio non può inviare pacchetti di lunghezza 0 (solo byte "dimensione")
28  if(lunghezza == 0) return Errore::inviaMessaggioVuoto;
29 
30 
31  // l'opzione 'insisti' permette di inviare anche quando un particolare stato
32  // della classe lo impedisce, aspettando, fino a un ragionavole timeout, che
33  // la condizione ostacolante sia risolta
34  if(!radioPronta(insisti)) return Errore::inviaTimeout;
35 
36  cambiaModalita(Modalita::standby, true);
37  disattivaAutoModes();
38 
39  // Il primo byte contiene la lunghezza del messaggio compresa l'intestazione
40  // ma sé stesso escluso.
41  // Anche le radio useranno questo valore per inviare/ricevere il pacchetto.
42  bus->scriviRegistro(RFM69_00_FIFO, lunghezza + 1);
43  // Il secondo byte è l'intestazione della classe
44  bus->scriviRegistro(RFM69_00_FIFO, intestazione);
45  // Tutti gli altri bytes sono il messaggio dell'utente
46  for(int i = 0; i < lunghezza; i++) {
47  bus->scriviRegistro(RFM69_00_FIFO, messaggio[i]);
48  }
49 
50 
51  // separa mesasggi con e senza richiesta di ACK
52  Intestazione intest;
53  intest.byte = intestazione;
54  if(intest.bit.richiestaAck) {
55  // metti la radio in modalità trasmissione con l'ordine di passare a
56  // ricezione non appena il pacchetto è stato inviato. In questo modo la
57  // radio è da subito pronta per ricevere un ACK. Siccome la radio non sembra
58  // essere capace di eseguire la sequenza ideale tx -> rx ->
59  // standby/modalitaDefault autonomamente, dopo la ricezione di un ACK rimane
60  // in modalità rx. (La condizione di uscita dalla modalità intermedia scelta
61  // (packetSent) non accade mai in modalità rx).
62  //
63  // Questa soluzione non è ottimale: per esempio un nuovo messaggio potrebbe
64  // essere sovrascritto all'ACK prima che questo sia riconosciuto da
65  // controlla(). Non si può però tornare alla modalità tx perché questo
66  // passaggio cancella la memoria FIFO, e non sembra esserci modo per passare
67  // a una terza modalità senza dover scrivere registri della radio nell'ISR.
68  autoModes(Modalita::tx, AMModInter::rx, AMEnterCond::packetSentRising, AMExitCond::crcOkRising);
69  stato = Stato::invioMessConAck;
70  statoUltimoAck = StatoAck::pendente;
71  }
72  else {
73  cambiaModalita(Modalita::tx);
74  stato = Stato::invioMessSenzaAck;
75  }
76 
77  tempoUltimaTrasmissione = millis();
78 
79  return Errore::ok;
80 
81 }

174 void RFM69::isr() {
175 
176  tempoUltimaEsecuzioneIsr = millis();
177 
178  switch(stato) {
179 
180  // Un interrupt in stato passivo può solo indicare la ricezione di un
181  // messaggio
182  case Stato::passivo:
183  // AutoModes si occupa di mettere la radio in standby
184  set(richiestaAzione.scaricaMessaggio);
185  set(richiestaAzione.inviaAckOTermina);
186  set(richiestaAzione.annunciaMessaggio);
187  stato = Stato::attesaAzione;
188  break;
189 
190  // fine della trasmissione
191  case Stato::invioMessSenzaAck:
192  ++messaggiInviati;
193  // siamo ancora in modalità tx, che non serve più
194  set(richiestaAzione.terminaProcesso);
195  stato = Stato::attesaAzione;
196  break;
197 
198  // fine della trasmissione
199  case Stato::invioMessConAck:
200  ++messaggiInviati;
201  // non è richiesta nessuna azione perché il passaggio da tx a rx
202  // necessario in questo momento è gestito autonomamente dalla radio
203  // grazie alla funtione AutoModes
204  stato = Stato::attesaAck;
205  break;
206 
207  // Questo caso si verifica se arriva un Ack (o un altro messaggio
208  // durante l'attesa di un ack)
209  case Stato::attesaAck:
210  statoUltimoAck = StatoAck::attesaVerifica;
211  set(richiestaAzione.scaricaMessaggio);
212  set(richiestaAzione.verificaAck);
213  set(richiestaAzione.terminaProcesso);
214  stato = Stato::attesaAzione;
215  break;
216 
217  case Stato::invioAck:
218  set(richiestaAzione.terminaProcesso);
219  break;
220 
221  default: break;
222  }

276 
277  // # 2. Esegui compiti ordinati dall'ISR per concludere un'azione #
278 
279  // nota: l'ordine di esecuzione è rilevante, perché alcune azioni dipendono
280  // dalla precedente esecuzione di altre

284  // riguardanti l'ACK (se il messaggio è un ack per verificarlo bisogna
285  // prima scaricarlo, se è un messaggio normale bisogna leggerne

303  ultimoRssi = -(bus->leggiRegistro(RFM69_24_RSSI_VALUE/2));

341  ++ackInattesi;
342  }
343  else {

357  if(modalitaDefault == Modalita::rx) {
358  // non serve disattivare AutoModes perché modalitaRicezione lo reimposta
359  modalitaRicezione();

365  }

369  // dall'utente (chiamando modalitaRicezione() mentre la radio è
370  // occupata). Per questo prima di rimuovere la propria flag controlla se

373  if(stato == Stato::passivo) {
374  clear(richiestaAzione.modalitaRxSePossibile);
375  modalitaRicezione();
376  }
377  }
378 
379  }

392 void RFM69::modalitaRicezione(bool aspetta) {
393  modalitaDefault = Modalita::rx;
394  if(radioPronta(aspetta)) {
395  // metti la radio in modalità rx con un'impostazione autoModes tale che
396  // appena un messaggio è ricevuto correttamente (crcOk) la radio passa
397  // in standby (quindi non riceve più).
398  // La condizione selezionata per uscire dallo standby non si verifica
399  // mai. Nello stesso momento in cui AutoModes cambia la modalità viene
400  // chiamata l'ISR, che segnala a `controlla()` l'arrivo di un messaggio.
401  autoModes(Modalita::rx, AMModInter::standby, AMEnterCond::crcOkRising, AMExitCond::packetSentRising);
402  }
403  else {
404  // La radio non è pronta per cambiare modalità -> delega a `controlla()`
405  set(richiestaAzione.modalitaRxSePossibile);
406  }

436 // Cambia la modalità di funzionamento
437 //
438 int RFM69::cambiaModalita(RFM69::Modalita mod, bool aspetta) {
439 
440  // Nota: la disattivazione e l'attivazione della modalità listen richiedono
441  // procedure speciali. Le aggiunte rispetto al codice per l'impostazione delle
442  // altre modalità sono marcate con "// *Listen*" (all'inizio di ogni blocco)
443 
444  // La modalità listen non è impostabile senza attesa
445  if(!aspetta && (mod == Modalita::listen)) return Errore::modImpossibile;
446 
447  // Se l'ISR ha appena chiesto una disattivazione del modo Tx potrebbe darsi
448  // che la radio è ancora in fase di transizione. 2 ms di attesa riducono
449  // le probabilità di conflitto (già molto bassa, e comunque non nulla).
450  // Al momento della scrittura di questa funzione, questo può succedere solo se
451  // l'impostazione TEMPO_MINIMO_FRA_MESSAGGI è 0 in caso di più invii di seguito
452  if(aspetta) delay(2);
453 
454  // Prepara il byte da scrivere nel registro
455  regOpMode &= 0xE3;
456  uint8_t codiceMod = 0x0;
457  switch(mod) {
458  case Modalita::sleep: codiceMod = 0x0; break;
459  // per entrare in modalità listen la radio deve essere in standby
460  case Modalita::listen:
461  case Modalita::standby: codiceMod = 0x1; break;
462  case Modalita::fs: codiceMod = 0x2; break;
463  case Modalita::tx: codiceMod = 0x3; break;
464  case Modalita::rx: codiceMod = 0x4; break;
465  }
466  regOpMode |= (codiceMod << 2);
467 
468  // *Listen* (1/3) - Disattiva
469  if (modalita == Modalita::listen) {
470  regOpMode &= !(1 << 6); // clear ListenOn
471  regOpMode |= 1 << 5; // set ListenAbort
472  }
473 
474  //Scrivi il registro RegOpMode
475  bus->scriviRegistro(RFM69_01_OP_MODE, regOpMode);
476 
477  // *Listen* (2/3) - Disattiva
478  if (modalita == Modalita::listen) {
479  regOpMode &= !(1 << 5); // clear ListenAbort
480  // Scrivi di nuovo RegOpMode senza il bit ListenAbort
481  bus->scriviRegistro(RFM69_01_OP_MODE, regOpMode);
482  }
483 
484 
485  // Impostazioni riguardanti la trasmissione high power (> 17dbM)
486 
487  // La radio deve entrare in modalità high power tx
488  else if (highPower && (mod == Modalita::tx)) highPowerSettings(true);
489  // La radio deve uscire dalla modalità high power tx
490  if (highPower && (modalita == Modalita::tx)) highPowerSettings(false);
491 
492  if(aspetta) {
493  // Aspetta che la radio sia pronta (questa flag è 0 durante il cambiamento di
494  // modalità)
495  // timeout: le transizioni più lunghe sembrano durare circa 1 ms (empirico)
496  unsigned long inizioAttesa = millis();
497  while(!(bus->leggiRegistro(RFM69_27_IRQ_FLAGS_1) & RFM69_FLAGS_1_MODE_READY)) {
498  if(inizioAttesa + 3 < millis()) return Errore::modTimeout;
499  }
500  }
501 
502  // *Listen* (3/3) - Attiva
503  // Per attivare la modalita listen basta scrivere il bit 6 dopo aver
504  // messo la radio in standby
505  if(mod == Modalita::listen) {
506  regOpMode |= 1 << 6;
507  bus->scriviRegistro(RFM69_01_OP_MODE, regOpMode);
508  }
509 
510  // Imposta l'interrupt generato sul pin DIO0
511  uint8_t dio0 = 0;
512  switch(mod) {
513  case Modalita::sleep: break; // non importa
514  case Modalita::listen: dio0 = 1; break;
515  case Modalita::standby: break; // non importa
516  case Modalita::fs: break; // non importa
517  case Modalita::tx: dio0 = 0; break;
518  case Modalita::rx: dio0 = 1; break;
519  }
520 
521  bus->scriviRegistro(RFM69_25_DIO_MAPPING_1, dio0 << 6);
522 
523  // Ricorda la modalita attuale della radio
524  modalita = mod;
525 
526  // switch(mod) {
527  // case Modalita::tx : Serial.println("_tx_"); break;
528  // case Modalita::rx : Serial.println("_rx_"); break;
529  // case Modalita::standby : Serial.println("_standby_"); break;
530  // case Modalita::sleep : Serial.println("_sleep_"); break;
531  // case Modalita::listen : Serial.println("_listen_"); break;
532  // default: break;
533  // }
534 
535  return Errore::ok;
536 }
