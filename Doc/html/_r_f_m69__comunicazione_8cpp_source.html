72 // [funzione privata] Invia un messaggio conoscendone già l'intestazione
73 //
74 int RFM69::inviaMessaggio(const uint8_t messaggio[], uint8_t lunghezza, uint8_t intestazione) {
75 
76  // la radio non può inviare pacchetti di lunghezza 0 (solo byte "dimensione")
77  if(lunghezza == 0) return Errore::inviaMessaggioVuoto;
78 
79  // Se c'è un messaggio in uscita non si può inviarne un altro. Visto che di
80  // solito la trasmissione dura pochi millisecondi, però, si può aspettare.
81  // (Se si ha fretta non bisogna chiamare invia due volte di seguito)
82  if(trasmissioneMessaggio || trasmissioneAck) {
83  uint32_t t = millis();
84  while(trasmissioneMessaggio || trasmissioneAck) {
85  // Il messaggio più lungo alla bitrate minima con l'encoding più
86  // dispendioso parte in una trentina di ms
87  if(millis() - 50 > t) return Errore::inviaTimeoutTxPrecedente;
88  }
89  }
90 
91  // Se la classe sta aspettando un ACK ignoralo (se l'utente sta inviando un
92  // nuovo messaggio significa che non controllerà più l'ack del precedente)
93  if(attesaAck) rinunciaAck();
94 
95 
96  cambiaModalita(Modalita::standby);
97 
98  // Il primo byte contiene la lunghezza del messaggio compresa l'intestazione
99  // ma sé stesso escluso.
100  // Anche le radio useranno questo valore per inviare/ricevere il pacchetto.
101  spi.scriviRegistro(RFM69_00_FIFO, lunghezza + 1);
102 
103  // Il secondo byte è l'intestazione della classe
104  spi.scriviRegistro(RFM69_00_FIFO, intestazione);
105 
106  // Tutti gli altri bytes sono il messaggio dell'utente
107  for(int i = 0; i < lunghezza; i++) {
108  spi.scriviRegistro(RFM69_00_FIFO, messaggio[i]);
109  }
110 
111  cambiaModalita(Modalita::tx);
112  trasmissioneMessaggio = true;
113 
114  Intestazione intest;
115  intest.byte = intestazione;
116  if(intest.bit.richiestaAck) attesaAck = true;
117  else attesaAck = false;
118  ackRicevuto = false;
119  tempoUltimaTrasmissione = millis();
120 
121  return Errore::ok;
122 
123 }

171 void RFM69::inviaAck() {
172  // Se c'è un messaggio in uscita non si può inviare un altro. Visto che di
173  // solito la trasmissione dura pochi millisecondi, però, si può aspettare.
174  // Alla fine dell'attesa il messaggio uscente viene bloccato.
175  if(trasmissioneMessaggio) {
176  uint32_t t = millis();
177  while(trasmissioneMessaggio) {
178  // Il messaggio più lungo alla bitrate minima con l'encoding più
179  // dispendioso parte in una trentina di ms
180  if(millis() - 50 > t) break;
181  }
182  }
183 
184  // Lunghezza, obbligatoria perché serve alla radio (1)
185  spi.scriviRegistro(RFM69_00_FIFO, 1);
186 
187  // Intestazione, segnala che il messaggio è un ACK
188  Intestazione intestazione;
189  intestazione.bit.ack = 1;
190  spi.scriviRegistro(RFM69_00_FIFO, intestazione.byte);
191 
192  cambiaModalita(Modalita::tx);
193  trasmissioneAck = true;
194 }

294 // Reagisce agli interrupt generati dalla radio sul suo pin DI0
295 //
296 void RFM69::isr() {
297  // ## Modalità TX, interrupt su `PacketSent` ## //
298 
299  // È appena stato trasmesso un messaggio
300  if(trasmissioneMessaggio) {
301  spi.usaInIsr(true);
302  if(attesaAck) cambiaModalita(Modalita::rx, false);
303  else cambiaModalita(modalitaDefault, false);
304  spi.usaInIsr(false);
305  trasmissioneMessaggio = false;
306  messaggiInviati++;
307  return;
308  }
309 
310  // È appena stato trasmesso un ACK
311  if(trasmissioneAck) {
312  spi.usaInIsr(true);
313  cambiaModalita(modalitaDefault, false);
314  spi.usaInIsr(false);
315  trasmissioneAck = false;
316  return;
317  }
318 
319 
320  // ## Modalità RX, interrupt su `PayloadReady` ## //
321 
322  if(modalita == Modalita::rx || modalita == Modalita::listen) {
323  // # Non si sa ancora se è un vero messaggio o solo un ACK #
324 
325  spi.usaInIsr(true);
326 
327  Modalita modPrec = modalita; // rx o listen
328  cambiaModalita(Modalita::standby, false);
329 
330  // Leggi i primi due bytes (lunghezza e intestazione)
331  uint8_t lung = spi.leggiRegistro(RFM69_00_FIFO);
332  Intestazione intest;
333  intest.byte = spi.leggiRegistro(RFM69_00_FIFO);
334 
335  if(attesaAck) {
336  // Il messaggio dovrebbe essere un ACK. È però possibile che l'altra
337  // radio invii a sua volta un messaggio prima di inviare l'ACK
338  // (l'ACK è inviato dalla funzione di lettura, chiamata dall'utente,
339  // mentre il messaggio è scaricato dalla FIFO della radio dall'ISR
340  // subito dopo la ricezione). Quindi se l'utente chiama prima invia()
341  // e poi leggi() sull'altra radio, questa ricevereà prima un messaggio
342  // e poi l'ACK atteso
343 
344  if(intest.bit.ack) {
345 
346  // # Il messaggio è un ACK # //
347 
348  attesaAck = false;
349  ackRicevuto = true;
350  cambiaModalita(modalitaDefault, false);
351  spi.usaInIsr(false);
352  // non c'è nient'altro da fare, un ACK non ha contenuto
353  return;
354  }
355  }
356 
357 
358  // # Il messaggio non è un ACK valido # //
359 
360  // Esiste anche la possibilità che un ACK arrivi inatteso (ad es. se
361  // l'utente ci ha rinunciato). In tal caso la radio deve restare in
362  // ascolto come se niente fosse, limitandosi a segnalare l'inutile ACK
363  // (un ACK inaspettato segnala che c'è un problema nel programma).
364  //
365  // Siccome è possibile che la modalità fosse listen la funzione
366  // `cambiaModalita()` non può essere usata in modo rapido (secondo
367  // argomento `false`).
368  else if(intest.bit.ack) {
369 
370  // # Il messaggio è un ACK indesiderato # //
371 
372  ackInattesi++;
373  cambiaModalita(modPrec, true); // modPrec potrebbe essere `listen`
374  spi.usaInIsr(false);
375  return;
376  }
377 
378  // # Il messaggio è un vero messaggio # //
379 
380  messaggioRicevuto = true;
381  messaggiRicevuti++;
382 
383  // leggi tutti gli altri bytes
384  spi.preparaTrasferimento();
385  spi.trasferisciByte(RFM69_00_FIFO);
386  for(int i = 0; i < lung - 1; i++) {
387  buffer[i] = spi.trasferisciByte();
388  }
389  spi.terminaTrasferimento();
390 
391  ultimoMessaggio.tempoRicezione = millis();
392  ultimoMessaggio.dimensione = lung - 1;
393  ultimoMessaggio.intestazione.byte = intest.byte;
394  ultimoMessaggio.rssi = -(spi.leggiRegistro(RFM69_24_RSSI_VALUE)/2);
395 
396  spi.usaInIsr(false);
397 
398  return;
399  }

411 // Cambia la modalità di funzionamento
412 //
413 int RFM69::cambiaModalita(RFM69::Modalita mod, bool aspetta) {
414 
415  // Nota: la disattivazione e l'attivazione della modalità listen richiedono
416  // procedure speciali. Le aggiunte rispetto al codice per l'impostazione delle
417  // altre modalità sono marcate con "// *Listen*" (all'inizio di ogni blocco)
418 
419  // La modalità listen non è impostabile senza attesa
420  if(!aspetta && (mod == Modalita::listen)) return Errore::modImpossibile;
421 
422  // Se l'ISR ha appena chiesto una disattivazione del modo Tx potrebbe darsi
423  // che la radio è ancora in fase di transizione. 2 ms di attesa riducono
424  // le probabilità di conflitto (già molto bassa, e comunque non nulla).
425  // Al momento della scrittura di questa funzione, questo può succedere solo se
426  // l'impostazione TEMPO_MINIMO_FRA_MESSAGGI è 0 in caso di più invii di seguito
427  if(aspetta) delay(2);
428 
429  // Prepara il byte da scrivere nel registro
430  uint8_t regOpMode = spi.leggiRegistro(RFM69_01_OP_MODE) & 0xE3;
431 
432  // Imposta i 3 bit che stabiliscono la modalità
433  uint8_t codiceMod;
434  switch(mod) {
435  case Modalita::sleep: codiceMod = 0x0; break;
436  // per entrare in modalità listen la radio deve essere in standby
437  case Modalita::listen:
438  case Modalita::standby: codiceMod = 0x1; break;
439  case Modalita::fs: codiceMod = 0x2; break;
440  case Modalita::tx: codiceMod = 0x3; break;
441  case Modalita::rx: codiceMod = 0x4; break;
442  }
443  regOpMode |= codiceMod << 2;
444 
445  // *Listen* (1/3) - Disattiva
446  if (modalita == Modalita::listen) {
447  regOpMode &= !(1 << 6); // clear ListenOn
448  regOpMode |= 1 << 5; // set ListenAbort
449  }
450 
451  //Scrivi il registro RegOpMode
452  spi.scriviRegistro(RFM69_01_OP_MODE, regOpMode);
453 
454  // *Listen* (2/3) - Disattiva
455  if (modalita == Modalita::listen) {
456  regOpMode &= !(1 << 5); // clear ListenAbort
457  // Scrivi di nuovo RegOpMode senza il bit ListenAbort
458  spi.scriviRegistro(RFM69_01_OP_MODE, regOpMode);
459  }
460 
461 
462  // Impostazioni riguardanti la trasmissione high power (> 17dbM)
463 
464  // La radio deve entrare in modalità high power tx
465  else if (highPower && (mod == Modalita::tx)) highPowerSettings(true);
466  // La radio deve uscire dalla modalità high power tx
467  if (highPower && (modalita == Modalita::tx)) highPowerSettings(false);
468 
469 
470  if(aspetta) {
471  // Aspetta che la radio sia pronta (questa flag è 0 durante il cambiamento di
472  // modalità)
473  unsigned long inizioAttesa = millis();
474  while(!spi.leggiRegistro(RFM69_27_IRQ_FLAGS_1) & RFM69_FLAGS_1_MODE_READY) {
475  if(inizioAttesa + 100 < millis()) return Errore::modTimeout;
476  }
477  }
478 
479  // *Listen* (3/3) - Attiva
480  // Per attivare la modalita listen basta scrivere il bit 6 dopo aver
481  // messo la radio in standby
482  if(mod == Modalita::listen) {
483  regOpMode = spi.leggiRegistro(RFM69_01_OP_MODE);
484  regOpMode |= 1 << 6;
485  spi.scriviRegistro(RFM69_01_OP_MODE, regOpMode);
486  }
487 
488  // Imposta l'interrupt generato sul pin DIO0
489  uint8_t dio0 = 0;
490  switch(mod) {
491  case Modalita::sleep: break; // non importa
492  case Modalita::listen: dio0 = 1; break;
493  case Modalita::standby: break; // non importa
494  case Modalita::fs: break; // non importa
495  case Modalita::tx: dio0 = 0; break;
496  case Modalita::rx: dio0 = 1; break;
497  }
498 
499  spi.scriviRegistro(RFM69_25_DIO_MAPPING_1, dio0 << 6);
500 
501  // Ricorda la modalita attuale della radio
502  modalita = mod;
503 
504  /*switch(mod) {
505  case Modalita::tx : Serial.print("_tx_"); break;
506  case Modalita::rx : Serial.print("_rx_"); break;
507  case Modalita::standby : Serial.print("_standby_"); break;
508  case Modalita::sleep : Serial.print("_sleep_"); break;
509  case Modalita::listen : Serial.print("_listen_"); break;
510  default: break;
511 }*/
512 
513 return Errore::ok;
514 }

570 //Imposta la potenza di trasmissione del segnale radio
571 //
572 bool RFM69::impostaPotenzaTx(int dBm) {
573 
574 
575  // Controlla che la potenza sia all'interno dei limiti
576  if(dBm < -18) return false;
577  if(dBm > 20) return false;
578 
579  bool pa0, pa1, pa2;
580  int outPow; //solo i primi 5 bit possono essere usati.
581 
582 
583  if(dBm <= -2) { //opzione 1
584  pa0 = 1;
585  pa1 = 0;
586  pa2 = 0;
587  outPow = dBm + 18;
588  highPower = false;
589  }
590  else if(dBm <= 13) { //opzione 2
591  pa0 = 0;
592  pa1 = 1;
593  pa2 = 0;
594  outPow = dBm + 18;
595  highPower = false;
596  }
597  else if(dBm <= 17) { //opzione 3
598  pa0 = 0;
599  pa1 = 1;
600  pa2 = 1;
601  outPow = dBm + 14;
602  highPower = false;
603  }
604  else if(dBm <= 20) { //opzione 4
605  pa0 = 0;
606  pa1 = 1;
607  pa2 = 1;
608  outPow = dBm + 11;
609  highPower = true;
610  }
611 
612  //prepara il byte che sarà scritto nel registro e invia alla radio
613  uint8_t paLevel = 0;
614  paLevel = (pa0 << 7) | (pa1 << 6) | (pa2 << 5) | outPow;
615 
616  spi.scriviRegistro(RFM69_11_PA_LEVEL, paLevel);
617 
618  return true;
619 }
