212

229  _monitor.print("break\n");

231  for(int i = 0; i < nrPuntini; i++) {

233  }

236  //-------------------------- 2a parte: attesa ------------------------------
237 
238  //numero di puntini che rappresentano lo scorrere del tempo `attesaMassima`
239  //già disegnati.

241

243  if (attesaMassima) {
244  _monitor.print("\n");
245  }
246 
247  //aspetta che l'utente o lo scadere del tempo massimo permettano di continuare
248  while(true) {
249
