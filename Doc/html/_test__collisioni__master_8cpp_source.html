219 void invia() {
220 
221  // Decidi caso se inviare o no, con una probabilità tale da avvicinarsi alla
222  // frequenza di invio `messAlMinuto`.
223 
224  // Tempo trascorso dall'ultima chiamata ad `invia()`, in microsecondi
225  uint32_t t = micros();
226  uint32_t deltaT = (t - microsInviaPrec);
227  microsInviaPrec = t;
228 
229  // `decisione` vale `true` con una probabilita di [messPerMin * deltaT / 1 min]
230  bool decisione = (messPerMin * deltaT > random(60000000));
231 
232  if(!decisione) return;
233 
234  novita = true;
235 
236  // crea un messaggio e inserisci la frequenza e poi numeri casuali
237  uint8_t mess[lunghezzaMessaggi];
238  mess[0] = messPerMin >> 8;
239  mess[1] = messPerMin;
240  for(int i = 2; i < lunghezzaMessaggi; i++) mess[i] = (uint8_t)(micros() % (500 - i));
241 
242  // Invia
243  accendiLed(LED_TX);
244  radio.inviaConAck(mess, lunghezzaMessaggi);
245  while(radio.ackInSospeso());
246 
247  if(radio.ricevutoAck()) {
248  messInviati++;
249  accendiLed(LED_ACK);
250  }
251  else {
252  messNonInviati++;
253  }
254 
255 }

348 void elaboraStatistiche() {
349 
350  // Calcolo messaggi al minuto
351  messPerMinEffettivi = ((uint32_t)(messInviati + messNonInviati)*60000)/(millis() - tInizio);
352 
353  // Calcolo percentuale di successo
354  // Nelle variabili sottostanti il valore 100 rappresenta un punto percentuale
355  if((messInviati + messNonInviati) > 0)
356  indiceSuccesso = (uint32_t)messInviati * 10000 / (messInviati + messNonInviati);
357 
358  // Determinazione della situazione di stabilità (usa il metodo Savitzky–Golay
359  // per calcolare la derivata della funzione del successo nel tempo)
360  deriv = 0;
361  if(nrElaborazioni >= 9) {
362  deriv += (float)indiciSuccessoPrec[0] * 86;
363  deriv += (float)indiciSuccessoPrec[1] * -142;
364  deriv += (float)indiciSuccessoPrec[2] * -193;
365  deriv += (float)indiciSuccessoPrec[3] * -126;
366  //deriv += indiciSuccessoPrec[4] * 0;
367  deriv += (float)indiciSuccessoPrec[5] * 126;
368  deriv += (float)indiciSuccessoPrec[6] * 193; // Usato come risultato finale
369  deriv += (float)indiciSuccessoPrec[7] * 142;
370  deriv += (float)indiceSuccesso * -86;
371  deriv /= 1188.0;
372  deriv /= 1000/(float)messPerMin;
373  if(deriv < 0) deriv = -deriv;
374 
375  if(deriv < (float)tolleranza/1000 && (millis() - tInizio) > durataMinimaTest)
376  statStabili = true;
377  }
378 
379 
380  // Aggiornamento dell'array
381  for(int i = 0; i < 7; i++)
382  indiciSuccessoPrec[i] = indiciSuccessoPrec[i+1];
383  indiciSuccessoPrec[7] = indiceSuccesso;
384 
385 
386  nrElaborazioni++;
387  if(millis() - tInizio > durataMinimaTest)
388  statStabili = true;
389 }
