106 // Prepara la comunicazione SPI.
107 // Il valore restituito deve essere passato come argomento alla funzione `terminaTransfer()`
108 //
109 void RFM69::Spi::preparaTrasferimento() {
110 
111  // Blocca gli interrupt
112  if(gestisciInterrupt) cli();
113 
114  // # Impedisci al pin SS (10 su Arduino UNO) di mettere improvvisamente SPI in
115  // modalità slave rendendolo temporeaneamente OUTPUT (se non lo è già)
116  // Attenzione: si tratta del pin SS, #defined da qualche parte da Arduino,
117  // non del pin ss, membro privato di questa classe!
118  // stato del pin #
119 
120  // Trova l'indirizzo del bit corrispondente a SS in DDR (Data Direction Register)
121  uint8_t bit = digitalPinToBitMask(SS); //SS, non ss
122  volatile uint8_t *reg = portModeRegister(digitalPinToPort(SS)); //SS, non ss
123 
124  // Se il bit non è impostato, la porta è usata come INPUT e potrebbe causare
125  // un cambio di modalità di SPI
126  if(!(*reg & bit)) {
127  pinSSInput = true;
128  pinMode(SS, OUTPUT); //SS, non ss
129  }
130  else pinSSInput = false;
131 
132 
133  // Inizia la transiszione
134  SPI.beginTransaction(SPISettings(frequenza, bitOrder, dataMode));
135 
136  // Attiva il pin SS scelto per comunicare con la radio (potrebbe essere anche
137  // quello trattato sopra)
138  digitalWrite(ss, LOW);
139 
140 }
