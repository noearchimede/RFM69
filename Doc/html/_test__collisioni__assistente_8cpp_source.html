136 void invia() {
137 
138 
139  // Decidi a caso se inviare o no, con una probabilità tale da avvicinarsi alla
140  // frequenza di invio `messAlMinuto`.
141 
142  // Tempo trascorso dall'ultima chiamata ad `invia()`, in microsecondi
143  uint32_t t = micros();
144  uint32_t deltaT = (t - microsInviaPrec);
145  microsInviaPrec = t;
146  // `decisione` vale `true` con una probabilita di [messPerMin * deltaT / 1 min]
147  bool decisione = ((messPerMin * deltaT) > random(60000000));
148 
149  if(!decisione) return;
150 
151  Serial.print("tx | mess/minuto: ");
152  Serial.println(messPerMin);
153 
154  // crea un messaggio e inserisci numeri casuali
155  uint8_t mess[lunghezzaMessaggi];
156  for(int i = 2; i < lunghezzaMessaggi; i++) mess[i] = (uint8_t)(micros() % (500 - i));
157 
158  // Invia
159  accendiLed(LED_TX);
160  radio.inviaConAck(mess, lunghezzaMessaggi);
161  while(radio.ackInSospeso());
162 
163 
164  if(radio.ricevutoAck()) accendiLed(LED_ACK);
165 
166 
167 }
