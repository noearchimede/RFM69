47 int RFM69::inviaFinoAck(uint16_t& tentativi, const uint8_t messaggio[], uint8_t lunghezza, uint8_t titolo) {
48 
49  Intestazione intestazione;
50  intestazione.bit.richiestaAck = 1;
51  if(titolo > valMaxTitolo) titolo = 0;
52  intestazione.bit.titolo = titolo;
53 
54  int errore;
55  uint16_t i = 0;
56  for(; i < tentativi; i++) {
57  // Invia il messaggio
58  errore = inviaMessaggio(messaggio, lunghezza, intestazione.byte);
59  if(errore != Errore::ok) return errore;
60  // attesa di al massimo `timeoutAck` millisecondi
61  while(ackInSospeso());
62  // controllo
63  if(ricevutoAck()) break;
64  }
65 
66  if(!ricevutoAck()) return Errore::errore;
67  tentativi = i;
68  return Errore::ok;
69 }
