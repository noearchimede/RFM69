212 // Inizializzazione della radio
213 // Cerca di inizializzare la radio e restituisce ErroreInit::ok (-> 0) se ci riesce
214 //
215 //
216 int RFM69::inizializza(uint8_t lunghezzaMaxMessaggio) {
217 
218 
219  // Questa versione della classe ha un'unica ISR, dichiarata come static,
220  // quindi in un programma ne può esistere una sola instanza.
221  if(nrIstanze > 1) return Errore::initTroppeRadio;
222 
223 
224  // ## SPI ## //
225 
226  // Inizializzazione di SPI
227  if(!spi.inizializza()) return Errore::initInitSPIFallita;
228 
229 
230  // ## RESET ## //
231 
232  // Il pin RESET della radio può facoltativamente essere collegato al microcontrollore
233  if(haReset) {
234  pinMode(pinReset, OUTPUT);
235  digitalWrite(pinReset, HIGH);
236  delay(10); //almeno 100 μs (microsecondi)
237  digitalWrite(pinReset, LOW);
238  delay(50);//almeno 5 ms (millisecondi)
239  }
240 
241  // ## ATTESA ATTIVAZIONE ## //
242 
243  // RFM69HCW richiede un'attesa di almeno 10 ms tra l'accensione e il primo
244  // messaggio. Questo delay li garantisce.
245  delay(50);
246 
247 
248  // ## CONTROLLO DELLA CONNESSIONE ## //
249 
250  // Il registo 0x10 della radio contiene la versione del chip. Per la radio
251  // per cui ho scritto questo programma (RFM69HCW ISM TRANSCEIVER MODULE v1.1)
252  // il valore è 0x24. Se il valore è differente non è garantito che questo
253  // programma funzioni.
254  uint8_t versioneChip = spi.leggiRegistro(RFM69_10_VERSION);
255 
256  // Se la versione risulata un byte uniforme, probabilmente nessun dispositivo
257  // ha risposto sul bus SPI
258  if(versioneChip == 0 || versioneChip == 0xff) return Errore::initNessunaRadioConnessa;
259 
260  // Se la versione non è quella attesa segnalalo e blocca.
261  else if (versioneChip != 0x24) return Errore::initVersioneRadioNon0x24;
262  // Se poi a un'analisi del programma il nuovo dispositivo risulterà compatibile
263  // con quello per cui ho scritto questo programma (versione == 0x24) basterà
264  // aggiungere qui sotto il nuovo numero come valore valido.
265 
266 
267  // ## INTERRUPT ## //
268 
269  // Preparazione dell'interrupt che gestià l'invio e la ricezione dei messaggi
270  uint8_t numeroInterrupt = digitalPinToInterrupt(pinInterrupt);
271 
272  // se il pin scelto come interrupt non ha questa capacità blocca l'inizializzazione
273  if(numeroInterrupt == NOT_AN_INTERRUPT) return Errore::initPinInterruptNonValido;
274 
275  // Ora il pin è per forza un interrupt; collegalo alla funzione ISR di questa classe
276  attachInterrupt(numeroInterrupt, isrCaller, RISING);
277 
278  // pointer a disposizione dell'ISR per riferire a questa classe
279  pointerRadio = this;
280 
281 
282  // ## IMPOSTAZIONE DELLA RADIO ## //
283 
284  if(!caricaImpostazioni()) return Errore::initErroreImpostazione;
285 
286  standby();
287 
288  // ## INIZIALIZZAZIONE DI VARIABILI ## //
289 
290  // # Allocazione di un'array da usare come buffer #
291 
292  // Il buffer potrebbe essere già stato allocato in una chimata precedente a
293  // questa funzione
294  if(buffer != nullptr) free(buffer);
295 
296  // Allocazione del (nuovo) buffer
297  buffer = (uint8_t*) malloc(sizeof(uint8_t) * lunghezzaMaxMessaggio);
298  lungMaxEntrata = lunghezzaMaxMessaggio;
299 
300  messaggiInviati = 0;
301  messaggiRicevuti = 0;
302 
303  return Errore::ok;
304 
305 }

310 // Scrittura in tutti i registri della radio dei valori definiti nel file di
311 // impostazione
312 //
313 bool RFM69::caricaImpostazioni() {
314 
315  // Se un pin di reset è disponibile esegui un reset della radio
316  if(haReset) {
317  digitalWrite(pinReset, HIGH);
318  delay(10); //almeno 100 μs (microsecondi)
319  digitalWrite(pinReset, LOW);
320  delay(50);//almeno 5 ms (millisecondi)
321  }
322 
323  // Scrivi tutti i valori dell'array valoreRegistri nei registri corrispondenti
324  int indice = 0;
325  int registro = 0;
326  while(registro < RFM69_ULTIMO_REGISTRO) {
327  if(!RFM69_RESERVED(registro)) {
328  spi.scriviRegistro(registro, VALORE_REGISTRI(indice));
329  indice++;
330  }
331  registro ++;
332  }
333 
334 
335  /*/ DEBUG
336  // Stampa valore effettivo e valore atteso (cioé scritto nell'array
337  // valoreRegistri() all'indice corrispondente) per ogni registro
338  int i = 0, r = 0;
339  while(r <= RFM69_ULTIMO_REGISTRO) {
340  if(!RFM69_RESERVED(r)) {
341  uint8_t reg = spi.leggiRegistro(r);
342  uint8_t val = VALORE_REGISTRI(i);
343  Serial.print("Reg 0x"); Serial.print(r, HEX);
344  Serial.print(": 0x"); Serial.print(reg, HEX);
345  Serial.print("\t");
346  Serial.print("Atteso (ind "); Serial.print(i);
347  Serial.print("): 0x"); Serial.print(val, HEX);
348  Serial.print("\t");
349  Serial.print(val == reg ? "" : "#");
350  Serial.print("\n");
351  i++;
352  }
353  r++;
354  }
355  // fine debug */
356 
357 
358  // Controlla un solo registro. Non è una prova certa della corretta scrittura
359  // di tutti i registri, ma poiché alcuni contengono anche byte il cui valore
360  // non è impostabile (unused, r, w, r/wc) una lettura di tutti i registri scritti
361  // richiederebbe molte eccezioni (es. leggi il registro 4 ignorando il bit 3)
362  //
363  // Il registro PaLevel non può avere valore 0x0 (non avrebbe senso) e tutti
364  // i suoi bits sono utilizzati e impostati da questo programma
365  // l'indice è 16 e non 17 perché c'è di mezzo un registro RESERVED
366  if(spi.leggiRegistro(RFM69_11_PA_LEVEL) != VALORE_REGISTRI(16)) return false;
367 
368  return true;
369 }

381 void RFM69::stampaRegistriSerial(HardwareSerial& serial) {
382 
383  bool lineaVuota = false;
384 
385  // l'ultimo registro è 0x71
386  for(int i = 0; i < 0x72; i++) {
387 
388  // Non leggere i registri riservati ma stampa al posto di ogni blocco di
389  // registri riservati una linea vuota
390  // La macro RFM69_RESERVED è definita in "Registri_RFM69.h"
391  if(RFM69_RESERVED(i)) {
392  if(!lineaVuota) serial.print("[...]\n");
393  lineaVuota = true;
394  continue;
395  }
396  else lineaVuota = false;
397 
398  //leggi il registro e stampane indirizzo e valore
399  uint8_t val = spi.leggiRegistro(i);
400  serial.print("0x"); serial.print(i,HEX);
401  serial.print("\t\t");
402  serial.print("0x"); serial.print(val, HEX);
403  serial.print("\n");
404  }
